67 double sptElapsedTime(const sptTimer timer) {
68  if(timer->use_cuda) {
69  return NAN;
70  } else {
71  return timer->stop_timespec.tv_sec - timer->start_timespec.tv_sec
72  + (timer->stop_timespec.tv_nsec - timer->start_timespec.tv_nsec) * 1e-9;
73  }
74 }
