124 int sptCudaMTTKRP(sptSparseTensor const * const X,
125  sptMatrix ** const mats, // mats[nmodes] as temporary space.

129  size_t const nmodes = X->nmodes;
130  size_t const nnz = X->nnz;
131  size_t const * const ndims = X->ndims;
132  size_t const R = mats[mode]->ncols;
133  size_t const stride = mats[mode]->stride;

139  if(mats[i]->ncols != mats[nmodes]->ncols) {
140  spt_CheckError(SPTERR_SHAPE_MISMATCH, "CUDA SpTns MTTKRP", "mats[i]->cols != mats[nmodes]->ncols");
141  }
142  if(mats[i]->nrows != ndims[i]) {

193  spt_CheckCudaError(result != 0, "CUDA SpTns MTTKRP");
