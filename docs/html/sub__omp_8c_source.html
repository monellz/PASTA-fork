24  /* Ensure X and Y are in same shape */
25  if(Y->nmodes != X->nmodes) {
26  spt_CheckError(SPTERR_SHAPE_MISMATCH, "OMP SpTns Sub", "shape mismatch");
27  }
28  for(size_t i = 0; i < X->nmodes; ++i) {
29  if(Y->ndims[i] != X->ndims[i]) {
30  spt_CheckError(SPTERR_SHAPE_MISMATCH, "OMP SpTns Sub", "shape mismatch");
31  }
32  }
33 
34  /* Determine partationing strategy. */
35  size_t * dist_nnzs_X = (size_t*)malloc(nthreads*sizeof(size_t));
36  size_t * dist_nnzs_Y = (size_t*)malloc(nthreads*sizeof(size_t));
37  size_t * dist_nrows_Y = (size_t*)malloc(nthreads*sizeof(size_t));
38 
39  spt_DistSparseTensor(Y, nthreads, dist_nnzs_Y, dist_nrows_Y);
40  spt_DistSparseTensorFixed(X, nthreads, dist_nnzs_X, dist_nnzs_Y);
41  free(dist_nrows_Y);
42 
43  printf("dist_nnzs_Y:\n");
44  for(size_t i=0; i<nthreads; ++i) {
45  printf("%zu ", dist_nnzs_Y[i]);
46  }
47  printf("\n");

51  }
52  printf("\n");
53  fflush(stdout);

56  /* Build a private arrays to append values. */
57  size_t nnz_gap = llabs((long long) Y->nnz - (long long) X->nnz);
58  size_t increase_size = 0;
59  if(nnz_gap == 0) increase_size = 10;
60  else increase_size = nnz_gap;
61 
62  sptSizeVector **local_inds = (sptSizeVector**)malloc(nthreads* sizeof *local_inds);
63  for(size_t k=0; k<nthreads; ++k) {
64  local_inds[k] = (sptSizeVector*)malloc(Y->nmodes* sizeof *(local_inds[k]));
65  for(size_t m=0; m<Y->nmodes; ++m) {
66  sptNewSizeVector(&(local_inds[k][m]), 0, increase_size);
67  }
68  }
69

71  for(size_t k=0; k<nthreads; ++k) {

73  }

75 
76  /* Add elements one by one, assume indices are ordered */
77  size_t Ynnz = 0;
78  omp_set_dynamic(0);
79  omp_set_num_threads(nthreads);
80  #pragma omp parallel reduction(+:Ynnz)
81  {
82  int tid = omp_get_thread_num();
83  size_t i=0, j=0;
84  Ynnz = dist_nnzs_Y[tid];
85  while(i < dist_nnzs_X[tid] && j < dist_nnzs_Y[tid]) {
86  int compare = spt_SparseTensorCompareIndices(X, i, Y, j);
87  if(compare > 0) {
88  ++j;
89  } else if(compare < 0) {
90  size_t mode;
91  int result;
92  for(mode = 0; mode < X->nmodes; ++mode) {
93  result = sptAppendSizeVector(&(local_inds[tid][mode]), X->inds[mode].data[i]);
94  spt_CheckOmpError(result, "OMP SpTns Sub", NULL);
95  }
96  result = sptAppendVector(&(local_vals[tid]), -X->values.data[i]);
97  spt_CheckOmpError(result, "OMP SpTns Sub", NULL);
98  ++Ynnz;
99  ++i;
100  } else {
101  Y->values.data[j] -= X->values.data[i];
102  ++i;
103  ++j;
104  }
105  }
106  /* Append remaining elements of X to Y */
107  while(i < dist_nnzs_X[tid]) {
108  size_t mode;
109  int result;
110  for(mode = 0; mode < X->nmodes; ++mode) {
111  result = sptAppendSizeVector(&(local_inds[tid][mode]), X->inds[mode].data[i]);
112  spt_CheckOmpError(result, "OMP SpTns Sub", NULL);
113  }
114  result = sptAppendVector(&(local_vals[tid]), -X->values.data[i]);
115  spt_CheckOmpError(result, "OMP SpTns Sub", NULL);
116  ++Ynnz;
117  ++i;
118  }
119 
120  }
121 
122  /* Append all the local arrays to Y. */
123  for(size_t k=0; k<nthreads; ++k) {
124  for(size_t m=0; m<Y->nmodes; ++m) {
125  sptAppendSizeVectorWithVector(&(Y->inds[m]), &(local_inds[k][m]));
126  }
127  sptAppendVectorWithVector(&(Y->values), &(local_vals[k]));
128  }
129 
130  for(size_t k=0; k<nthreads; ++k) {
131  for(size_t m=0; m<Y->nmodes; ++m) {
132  sptFreeSizeVector(&(local_inds[k][m]));
133  }
134  free(local_inds[k]);
135  sptFreeVector(&(local_vals[k]));
136  }
137  free(local_inds);
138  free(local_vals);
139  free(dist_nnzs_X);
140  free(dist_nnzs_Y);
141 
142  /* Check whether elements become zero after adding.
143  If so, fill the gap with the [nnz-1]'th element.
144  */
145  spt_SparseTensorCollectZeros(Y);
146  /* Sort the indices */
147  sptSparseTensorSortIndex(Y);
148  return 0;
149 }
