75 double sptElapsedTime(const sptTimer timer) {
76  if(timer->use_cuda) {
77  float elapsed;
78  if(cudaEventElapsedTime(&elapsed, timer->start_event, timer->stop_event) != 0) {
79  return NAN;
80  }
81  return elapsed * 1e-3;
82  } else {
83  return timer->stop_timespec.tv_sec - timer->start_timespec.tv_sec
84  + (timer->stop_timespec.tv_nsec - timer->start_timespec.tv_nsec) * 1e-9;
85  }
86 }
