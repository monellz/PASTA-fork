37 static void spt_QuickSortAtMode(sptSparseTensor *tsr, size_t l, size_t r, size_t mode) {
38  size_t i, j, p;
39  if(r-l < 2) {
40  return;
41  }
42  p = (l+r) / 2;
43  for(i = l, j = r-1; ; ++i, --j) {
44  while(spt_SparseTensorCompareAtMode(tsr, i, tsr, p, mode) < 0) {
45  ++i;

48  --j;
